﻿#include<filesystem>#include<fstream>#include<iostream>usingnamespacestd;namespacefs=filesystem;intmain(){vector<string>ext;stringbuffer;ifstreamconfig("config.txt");if(!config){cout<<"Confignotfound.Search*.c,*.cppand*.h\n";ext={".c",".cpp",".h"};}else{while(config>>buffer)if(!buffer.empty()&&buffer[0]!='#'&&buffer[0]!=';')ext.push_back(buffer);}vector<string>out;for(auto&file:fs::recursive_directory_iterator(fs::current_path())){if(find(ext.begin(),ext.end(),file.path().extension())!=ext.end()){ifstreaminfile(file.path());if(infile){out.push_back("//=========="+file.path().filename().string()+"==========\n");while(infile>>buffer)out.push_back(buffer);infile.close();}}}ofstreamoutfile("out.txt");if(!outfile){cerr<<"Errorcreatingout.txt\n";system("pause");}for(auto&s:out)outfile<<s;outfile.close();cout<<"Done.\n";system("pause");}
